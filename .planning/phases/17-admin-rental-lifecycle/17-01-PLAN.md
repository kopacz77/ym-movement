    ## File header (imports + module-private helpers)

    ```typescript
    import { PaymentMethod, type Prisma, RentalRequestStatus, RentalType } from "@prisma/client";
    import { TRPCError } from "@trpc/server";
    import { addDays } from "date-fns";
    import { z } from "zod";
    import { createNotification } from "@/features/notifications/utils/notificationHelpers";
    import { adminProcedure, createTRPCRouter } from "@/lib/trpc";
    import { getWardrobeSettings } from "./wardrobeSettingsQueries";

    /**
     * Maps RentalType to the dress price column that should be snapshotted as Rental.rentalFee.
     * Throws BAD_REQUEST if the caller requested PURCHASE but the dress has no purchase price.
     */
    function pickRentalFee(
      dress: { competitionPrice: number; seasonalPrice: number; purchasePrice: number | null },
      type: RentalType,
    ): number {
      switch (type) {
        case RentalType.COMPETITION:
          return dress.competitionPrice;
        case RentalType.SEASONAL:
          return dress.seasonalPrice;
        case RentalType.PURCHASE:
          if (dress.purchasePrice == null) {
            throw new TRPCError({
              code: "BAD_REQUEST",
              message: "This dress is not offered for purchase",
            });
          }
          return dress.purchasePrice;
      }
    }

    /**
     * RENTAL-03 (LOCKED): consignmentCommissionPct === 0 means Yura/platform-owned → payout null.
     * consignmentCommissionPct > 0 means consigned → payout populated.
     * Math.round used (matches JS half-up convention; document any audit deltas against banker's rounding).
     */
    function computeConsignmentPayout(
      dress: { consignmentCommissionPct: number },
      rentalFee: number,
    ): number | null {
      if (dress.consignmentCommissionPct === 0) return null;
      return rentalFee - Math.round((rentalFee * dress.consignmentCommissionPct) / 100);
    }
    ```

    ## wardrobeRequestRouter (3 procedures)

    ### `listRequests`
    Input: `{ status?: z.nativeEnum(RentalRequestStatus), page?: number = 1, limit?: number = 20 }`.
    Default `where.status = "PENDING"` when input.status is undefined.
    Include `Dress: { select: { id, title, sizeLabel, color, competitionPrice, seasonalPrice, purchasePrice, cleaningFee, securityDeposit, consignmentCommissionPct, status, Images: { where: { isPrimary: true }, select: { url: true }, take: 1 } } }` and `Student: { select: { id, User: { select: { name, email } } } }`.
    `orderBy: [{ competitionDate: { sort: "asc", nulls: "last" } }, { createdAt: "asc" }]`.     Return: `{ requests, total, page, limit }` (mirror `wardrobeDressQueries.list`).

    ### `respondToRequest`
    Input: `z.object({ requestId: z.string().cuid(), decision: z.enum(["APPROVE", "DECLINE"]), responseMessage: z.string().min(1).max(1000) })`.
    Wrap in `ctx.prisma.$transaction(async (tx) => {...})`:
    1. Fetch request with `select: { id, status, expiresAt, dressId, studentId, Dress: { select: { title, status, ownerId } }, Student: { select: { User: { select: { id, name } } } } }`.
    2. If !request → NOT_FOUND.
    3. If `request.status !== "PENDING"` → BAD_REQUEST "Only PENDING requests can be responded to".
    4. If `request.expiresAt && request.expiresAt < new Date()` → BAD_REQUEST "Request has expired".
    5. If APPROVE: assert `request.Dress.status === "AVAILABLE"`; otherwise BAD_REQUEST "Dress is already held by another approved request" (LOCKED per research Open Question 1).
    6. Compute `newRequestStatus = APPROVE ? "APPROVED" : "DECLINED"`; `newDressStatus = APPROVE ? "PENDING" : "AVAILABLE"`.
    7. `tx.rentalRequest.update({ where: { id: request.id }, data: { status: newRequestStatus, ownerResponse: input.responseMessage, respondedAt: new Date() } })`.
    8. `tx.dress.update({ where: { id: request.dressId }, data: { status: newDressStatus } })`.
    9. Return `{ request, decision: input.decision }`.

    AFTER tx commits, fire `createNotification` to `request.Student.User.id` with title/message tailored to APPROVE vs DECLINE, link `/wardrobe/my-rentals`, type `SUCCESS` (APPROVE) or `WARNING` (DECLINE). Wrap in `try { ... } catch (err) { console.error("[WARDROBE] Failed to notify student:", err); }` — non-blocking.

    Final return: `{ id: request.id, decision }`.

    ### `markPaymentReceived`
    Input: `z.object({ requestId: z.string().cuid(), paymentMethod: z.nativeEnum(PaymentMethod) })`.
    Wrap in `ctx.prisma.$transaction(async (tx) => {...})`:
    1. Fetch request with `select: { id, status, dressId, studentId, rentalType, startDate, endDate, Dress: { select: { id, title, status, competitionPrice, seasonalPrice, purchasePrice, cleaningFee, securityDeposit, consignmentCommissionPct } }, Student: { select: { User: { select: { id, name } } } } }`.
    2. If !request → NOT_FOUND.
    3. If `request.status !== "APPROVED"` → BAD_REQUEST "Only APPROVED requests can be converted".
    4. If `request.Dress.status !== "PENDING"` → BAD_REQUEST "Dress is not in PENDING state".
    5. `const rentalFee = pickRentalFee(request.Dress, request.rentalType);`
    6. `const consignmentPayoutAmount = computeConsignmentPayout(request.Dress, rentalFee);`
    7. `const totalCharged = rentalFee + request.Dress.cleaningFee + request.Dress.securityDeposit;`
    8. `tx.rentalRequest.update({ where: { id: request.id }, data: { status: "CONVERTED" } })`.
    9. `const created = await tx.rental.create({ data: { dressId, studentId, requestId, rentalType, startDate, endDate, rentalFee, cleaningFee: request.Dress.cleaningFee, securityDeposit: request.Dress.securityDeposit, totalCharged, paymentMethod: input.paymentMethod, paymentStatus: "PAID", depositCollectedAt: new Date(), consignmentPayoutAmount } })`.
    10. `tx.dress.update({ where: { id: request.dressId }, data: { status: "RENTED" } })`.
    11. Return `{ created, request }`.

    AFTER tx commits, fire `createNotification` (type SUCCESS, title "Payment confirmed", link `/wardrobe/my-rentals`) wrapped in try/catch.

    Return `{ id: created.id }`.
