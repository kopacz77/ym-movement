    ## Component body

    ```typescript
    const utils = api.useUtils();
    const form = useForm<PaymentInput>({
      resolver: zodResolver(paymentSchema),
      defaultValues: { paymentMethod: PaymentMethod.VENMO },
    });

    const markPaid = api.admin.wardrobeRequests.markPaymentReceived.useMutation({
      onSuccess: () => {
        utils.admin.wardrobeRequests.listRequests.invalidate();
        utils.admin.wardrobeRentals.listRentals.invalidate();
        toast.success("Rental confirmed", {
          description: `${studentName}'s rental for "${dressTitle}" is now active.`,
        });
        onOpenChange(false);
        form.reset({ paymentMethod: PaymentMethod.VENMO });
      },
      onError: (e) => toast.error("Failed to record payment", { description: e.message }),
    });

    const onSubmit = (data: PaymentInput) => {
      markPaid.mutate({ requestId, paymentMethod: data.paymentMethod });
    };
    ```

    const form = useForm<ReturnedInput>({
      resolver: zodResolver(returnedSchema),

    type DialogState =

      onSuccess: () => {

    ## Returns-due bucket derivation
    ```typescript
    const returnReminderDays = settingsQuery.data?.wardrobeReturnReminderDays ?? 1;
    const now = new Date();
    const dueByThreshold = addDays(now, returnReminderDays);

    const allActive = activeQuery.data?.rentals ?? [];
    const returnsDue = allActive.filter(
      (r) => r.paymentStatus === "PAID" && isBefore(new Date(r.endDate), dueByThreshold),
    );
    const others = allActive.filter((r) => !returnsDue.some((d) => d.id === r.id));
    ```

    ## Confirmation handlers

    const handleReleaseDeposit = (r: ActiveRental) => {
      // Use showDeleteConfirmation (or showConfirmationToast — whichever the helper exports as "destructive" pattern)

        title: "Release deposit",
        description: `Release security deposit (${formatCurrencyFromCents(r.securityDeposit)}) for ${r.Student.User.name}'s rental of "${r.Dress.title}"? The dress will return to AVAILABLE.`,

        confirmLabel: "Flag for Late Fee",
        onConfirm: () => flagLateFee.mutate({ rentalId: r.id }),
      });
    };
