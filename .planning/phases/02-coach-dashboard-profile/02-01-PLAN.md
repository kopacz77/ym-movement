             .then((userData) => {
               if (isMounted) {
                 if (userData.Coach?.id) {
                   setCoachId(userData.Coach.id);
                   setIsApproved(userData.Coach.isApproved ?? false);

                   setIsApproved(false);
                   setIsActive(false);
                 }
               }
             })

               setIsApproved(false);
               setIsActive(false);

         else if (session?.user?.role === "ADMIN" || session?.user?.role === "SUPER_ADMIN") {
